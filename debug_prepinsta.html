printf("%d ", a[i][right]);

right--;

the remaining rows */

printf("%d ", a[bottom][i]);
}
bottom--;
}

/* Print the first column from
the remaining columns */
if (left <= right) { for (i = bottom; i >= top; --i) {
printf("%d ", a[i][left]);
}
left++;
}
}

return 0;
}
